322 /* Normalize is an enum to handle different histogram normalization schemes
323  *
324  * NO: don't perform any normalization, output the binned weight verbatim
325  * INTEGRATABLE: divide bin weight by bin width
326  * UNITARY: divide bin weight by total binned weight (including under/over-flow)
327  * PDF: both INTEGRATABLE and UNITARY (i.e. like a probability distribution)
328  * NOTE: PDF = INTEGRATABLE & UNITARY
329  * (i.e. it's not 3 simply because 3 comes after 2)
330 */
331 enum class Normalize : norm_t {NO = 0, INTEGRATABLE = 1, UNITARY = 2, PDF = 3};
